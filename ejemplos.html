<!-- ↓ Reemplaza con tu código real ↓ -->
<section class="hero">
  <h1 id="typewriter"></h1>
  <p>Estrategia Digital & Automatización</p>
  <a href="contacto.html" class="btn">Contáctame</a>
</section>

<script>
  const words = ["Marketing Digital", "Automatización CRM", "Desarrollo Web"];
  let w = 0, c = 0, deleting = false;
  function type() {
    const el = document.getElementById("typewriter");
    const word = words[w % words.length];
    el.textContent = deleting ? word.slice(0, --c) : word.slice(0, ++c);
    if (!deleting && c === word.length) { deleting = true; setTimeout(type, 1200); return; }
    if (deleting && c === 0) { deleting = false; w++; }
    setTimeout(type, deleting ? 60 : 100);
  }
  type();
</script>

Automatización de Reportes con Python

Python · openpyxl · smtplib
Python

Script que genera un reporte Excel automatizado y lo envía por correo cada lunes al equipo de marketing.

Ver en GitHub
reporte.py 
# ↓ Reemplaza con tu código real ↓
import openpyxl, smtplib, schedule, time
from email.mime.multipart import MIMEMultipart
from email.mime.base import MIMEBase
from email import encoders

def generar_reporte():
    wb = openpyxl.Workbook()
    ws = wb.active
    ws.title = "Reporte Semanal"
    ws.append(["Campaña", "Impresiones", "Clics", "CTR"])
    # → Aquí conectas tu fuente de datos real
    wb.save("reporte_semanal.xlsx")
    print("Reporte generado.")

def enviar_email():
    msg = MIMEMultipart()
    msg["Subject"] = "Reporte Semanal Marketing"
    # → Configurar credenciales SMTP en variables de entorno
    print("Email enviado.")

schedule.every().monday.at("08:00").do(lambda: (generar_reporte(), enviar_email()))
while True:
    schedule.run_pending()
    time.sleep(60)
